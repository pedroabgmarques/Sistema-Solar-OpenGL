			precoEst -= 300;

			else
				precoEst += precoEst / 100;

		if (precoEst >= 10000)
			precoEst += precoEst / 10;
		else
			precoEst += precoEst / 7;
	}
	if (precoEst<300)
	{
		precoEst = 300;
	}

	return precoEst;
